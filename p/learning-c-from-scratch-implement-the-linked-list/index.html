1.数据结构介绍

1.1什么是数据结构？

数据结构是计算机的内存中存储和组织数据的方式

数据结构关注如何以最有效的方式组织和存储数据，以便在计算机程序

程序中进行操作和处理

1.2为什么需要数据结构？

在日常生活中，餐厅里如果客人很多，他们就会将客人的订单

排列好，比如先上1号桌的菜、再上2号桌…，这样就不会出现客户

就餐体验差、等餐时间长、餐厅营业混乱等情况

同理，程序如果不对数据进行管理，可能会导致数据丢失、操作数据困

难、野指针等情况。

通过数据结构，能够有效将数据组织和管理在一起。

按照我们的方式任意对数据进行增、删、改、查等操作。

1.3常见数据结构分类

一般按照组织形式，分为：线性数据结构和非线性数据结构

2.链表

2.1概念

链表是一种线性的数据结构，由一系列的结点组成。

每个结点包含两个部分：数据和指向下一个节点的指针

链表中的元素在内存中不一定是顺序排列的，而是通过指针相互

连接

#include <iostream>
using namespace std;

//声明一个节点的类型
struct Node
{
	int data;
    Node* next;
};

int main()
{
	struct Node n1 = {1, NULL};
	struct Node n2 = { 2, NULL };
	struct Node n3 = { 3, NULL };
	struct Node n4 = { 4, NULL };

	n1.next = &n2;
	n2.next = &n3;
	n3.next = &n4;

	return 0;
}

2.3链表的分类

  • 单向：每个节点只有一个指针指向下一个节点
  • 双向：每个节点有两个指针，分别指向前一个节点和后一个节点
  • 循环：尾结点指向头结点
  • 不循环

2.4单链表的实现

动态申请链表结点和链表构建

我们前面实现的链表的节点都是局部变量，局部变量是存放在栈区的，

生命周期比较短，出作用域就系统自动回收所在空间。

但是我们希望创建一个链表，存好数据后，在我不希望这个节点回收的时候，

它就不能回收，它得保持一直存在。（出了作用域也得存在）

所以，我们可以使用之前学过的动态内存管理的知识，使用new和delete

new是在堆区申请空间，申请后会返回对应空间的地址

void PushBack(Node*& phead, int d)
{
	//1.创建新节点
	Node* newnode = CreateNode(d);
	if (phead == NULL)//空链表
	{
		phead = newnode;
	}
	else//非空链表
	{
		//2.找尾结点
		Node* ptail = phead;
		while (ptail->next)
		{
			ptail = ptail->next;
		}
		//3.让尾结点的next指针指向新节点
		ptail->next = newnode;
	}
}

void PopBack(Node*& phead)
{
	if (phead == NULL)//链表为空
	{
		cout << "链表为空，不可删除" << endl;
	}
	else//链表不为空
	{
		if (phead->next == NULL)//链表只有一个节点
		{
			delete phead;
			phead = NULL;
		}
		else
		{
				Node* prev = NULL;
				Node* tail = phead;
				while (tail->next)
				{
					prev = tail;
					tail = tail->next;
				}
				//释放tail节点
				delete tail;
				tail = NULL;
				//前一个节点prev的next置为空
				prev->next = NULL;
			}
		}
}

单链表删除所有元素

链表为空，不可销毁

链表不为空

  1. 定义一个指针pcur指向头结点，和next指向pcur的下一个节点

  2. 让pcur遍历链表并释放pcur所指向的节点，然后pcur = next，next = next->next

void DestroyList(Node*& phead)
{
	if (phead == NULL)
	{
		cout << "链表为空，不可销毁" << endl;
	}
	else
	{
		Node* next = phead->next;
		while (phead)
		{
			delete phead;
			phead = next;
			if(next)
				next = next->next;
		}
		next = phead = NULL;
	}
}

#include <iostream>
using namespace std;

//声明一个节点的类型
struct Node
{
	int data;
	Node* next;
};

//链表
class List
{
private:
	//成员变量
	Node* phead; // 指向第一个节点的指针

public:
	//成员函数
	//构造函数
	List()
	{
		cout << "调用构造函数" << endl;
		phead = NULL;
	}
	//析构函数 - 释放链表
	~List()
	{
		cout << "调用析构函数" << endl;
		while (phead)
		{
			PopFront();
		}
	}

	//创建节点
	Node* CreateNode(int d)
	{
		Node* pn = new Node;
		pn->data = d;
		pn->next = NULL;

		return pn;
	}

	//链表元素的打印
	void PrintList()
	{
		Node* cur = phead;
		while (cur)
		{
			cout << cur->data << " ";
			cur = cur->next;
		}
		cout << endl;
	}

	//头插
	void PushFront(int d)
	{
		Node* newNode = CreateNode(d);
		newNode->next = phead;
		phead = newNode;
	}

	//尾插
	void PushBack(int d)
	{
		//1.创建新节点
		Node* newnode = CreateNode(d);
		if (phead == NULL)//空链表
		{
			phead = newnode;
		}
		else//非空链表
		{
			//2.找尾结点
			Node* ptail = phead;
			while (ptail->next)
			{
				ptail = ptail->next;
			}
			//3.让尾结点的next指针指向新节点
			ptail->next = newnode;
		}
	}

	//头删
	void PopFront()
	{
		if (phead)
		{
			Node* next = phead->next;
			delete phead;
			phead = next;
		}
		else
		{
			cout << "链表为空，无法删除" << endl;
		}
	}

	//尾删
	void PopBack()
	{
		if (phead == NULL)//链表为空
		{
			cout << "链表为空，不可删除" << endl;
		}
		else//链表不为空
		{
			if (phead->next == NULL)//链表只有一个节点
			{
				delete phead;
				phead = NULL;
			}
			else
			{
				Node* prev = NULL;
				Node* tail = phead;
				while (tail->next)
				{
					prev = tail;
					tail = tail->next;
				}
				//释放tail节点
				delete tail;
				tail = NULL;
				//前一个节点prev的next置为空
				prev->next = NULL;
			}
		}
	}
};

int main()
{
	List list1;//创建了一个空链表
	list1.PushBack(1);
	list1.PushBack(2);
	list1.PushBack(3);
	list1.PushBack(4);

	list1.PrintList();//1 2 3 4
	return 0;
}
