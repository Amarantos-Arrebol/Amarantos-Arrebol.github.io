1.序列容器和关联式容器

序列式容器

逻辑结构为线性序列的数据结构，两个位置存储的值之间一般没有紧密的关联关系

比如交换一下，他依旧是序列式容器

顺序容器中的元素是按他们在容器中的存储位置来顺序保存和访问的

序列式容器有 string、vector、list、deque、array、forward_list 等

关联式容器

关联式容器逻辑结构通常是非线性结构，两个位置有紧密的关联关系

比如交换一下，他的存储结构就被破坏了

顺序容器中的元素是按关键字来保存和访问的

关联式容器有 map/set(底层红黑树)系列和 unordered_map/unordered_set(底层哈希表) 系列

map 和 set 底层是红黑树，红黑树是一颗平衡二叉搜索树

  • set 里面存的是一个单独的关键字

  • map 里面存的是一个 pair<key, value>（k-v 模型），不仅有一个关键字，还会有一个与关键字绑定的值，比较方式是按照 key 的值来比较（排序是按照第一个关键字来比较的）

  • set 和 map 不能存重复的元素，multiset 和 multimap 可以存重复的元素

2.set / multiset

2.1 基本介绍

set 底层是用红黑树实现，增删查效率是 O(logN）

迭代器遍历是走的搜索树的中序，因此是一个有序序列

set 不能存重复元素，multiset 可以存重复元素，所以可以用 set 来去重

文档

<set 底层关键字的类型，仿函数，空间配置器>

（1）一般情况下，我们只需要传第一个模版参数

（2）需要传仿函数的场景：

T 不支持比较大小，需要自己写仿函数来支持

T 支持比较大小，但不符合我们的需求

（3）set 底层存储数据的内存是从空间配置器申请的，如果需要，可以自己实现内存池传给 set 的第三个参数

如果想查找元素是否在 set 中，我们一般不使用 find，而是 count。因为 find 的返回值是一个迭代器，判断起来不方便

lower_bound 和 upper_bound

**lower_bound：**大于等于 x 的最小元素，返回的是迭代器。时间复杂度：O（logN）

**upper_bound：**大于 x 的最小元素，返回的是迭代器。时间复杂度：O(logN)

使用场景：删掉某个迭代器区间

由于迭代器区间一般要求左闭右开，所以 lower_bound 的返回值可以作为"左闭"，upperbound 的返回值可以作为"右开"

	for (auto e : myset)

	}

2.4 multiset 的使用

multiset 和 set 的使用基本完全一致。

multiset 和 set 的区别：

  • multiset 里面可以存重复的元素，所以元素放到 multiset 里面相当于只排序了，没有去重。

  • multiset 中 erase 删除时如果传的是某个元素，那么会将 multiset 中所有等于该元素的值都删掉

  • multiset 中 find 查找的值 x 可能有多个，返回的是中序的第一个

#include <iostream>
#include <set>

using namespace std;

int main()
{
	//multiset会排序，但不去重
	multiset<int> mp = { 4, 2, 7, 4, 8, 4, 5, 4, 9 };
	for (auto e : mp)
	{
		cout << e << " ";
	}
	cout << endl;

	//multiset中x可能存在多个，find找的是中序的第一个
	int x; cin >> x;
	auto pos = mp.find(x);
	while (pos != mp.end() && *pos == x)
	{
		cout << *pos << " ";
		++pos;
	}
	cout << endl;

	//multiset中x可能存在多个，erase会删除所有的x
	mp.erase(x);
	for (auto e : mp)
	{
		cout << e << " ";
	}
	cout << endl;

	return 0;
}

class Solution {
public:
    vector<int> intersection(vector<int>& nums1, vector<int>& nums2) {
        //排序+去重
        set<int> mp1(nums1.begin(), nums1.end());
        set<int> mp2(nums2.begin(), nums2.end());

        //依次比较，小的++，相等的就是交集
        vector<int> ret;
        auto it1 = mp1.begin(), it2 = mp2.begin();
        while(it1 != mp1.end() && it2 != mp2.end())
        {
            if(*it1 < *it2) it1++;
            else if(*it1 > *it2) it2++;
            else
            {
                ret.push_back(*it1);
                it1++; it2++;
            }
        }

        return ret;
    }
};

补充：算法库中提供了求并集、交集、差值等集合运算的方法

求并集、交集，差集的思路

  • 求并集：放到 set 中，最后留下来的就是并集

  • 求交集：

    依次比较

    (1）小的++

    (2）相等的就是交集，同时++

    (3）其中一个结束就结束了

  • 求差集：

    依次比较

    (1）小的就是差集，小的++

    (2）相等就同时++

    (3）其中一个结束就结束了

应用场景：本地和云端的同步

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode *detectCycle(ListNode *head) {
        set<ListNode*> mp;
        ListNode* cur = head;
        while(cur)
        {
            auto ret = mp.insert(cur);
            if(ret.second == false) return cur;

            cur = cur->next;
        }
        return nullptr;
    }
};

3.1 基本介绍

文档

class Solution {
public:
    Node* copyRandomList(Node* head) {
        map<Node*, Node*> nodeMap;
        Node* copyhead = nullptr, *copytail = nullptr;

        Node* cur = head;
        while(cur)
        {
            if(copytail == nullptr)
            {
                copyhead = copytail = new Node(cur->val);
            }
            else 
            {
                copytail->next = new Node(cur->val);
                copytail = copytail->next;
            }

            //原结点和拷贝结点map kv存储
            nodeMap[cur] = copytail;

            cur = cur->next;
        }

        //处理random
        cur = head;
        Node* copy = copyhead;
        while(cur)
        {
            if(cur->random == nullptr)
            {
                copy->random = nullptr;
            }
            else 
            {
                copy->random = nodeMap[cur->random];
            }

            cur = cur->next;
            copy = copy->next;
        }
        return copyhead;
    }
};

前 K 个高频单词

思路：

用排序找前k个单词，因为map中已经对key单词排序过，也就意味着遍历map时，次数相同的单词，字典序小的在前面，字典序大的在后面。那么我们将数据放到vector中用一个稳定的排序就可以实现上面特殊要求，但是sort底层是快排，是不稳定的，所以我们要用stable_sort，他是稳定的。

class Solution {
public:
    struct Compare
    {
        bool operator()(const pair<string, int>& kv1, const pair<string, int>& kv2)
        {
            return kv1.second > kv2.second;
        }
    };

    vector<string> topKFrequent(vector<string>& words, int k) {
        map<string, int> countMap;
        for(auto& e : words)
        {
            countMap[e]++;
        }

        vector<pair<string, int>> v(countMap.begin(), countMap.end());
        stable_sort(v.begin(), v.end(), Compare());

        vector<string> retv;
        for(size_t i = 0; i < k; i++)
        {
            retv.push_back(v[i].first);
        }

        return retv;
    }
};

class Solution {
public:
    struct Compare
    {
        bool operator()(const pair<string, int>& kv1, const pair<string, int>& kv2)
        {
            return kv1.second > kv2.second
            || (kv1.second == kv2.second && kv1.first < kv2.first);
        }
    };

    vector<string> topKFrequent(vector<string>& words, int k) {
        map<string, int> countMap;
        for(auto& e : words)
        {
            countMap[e]++;
        }

        vector<pair<string, int>> v(countMap.begin(), countMap.end());
        sort(v.begin(), v.end(), Compare());

        vector<string> retv;
        for(size_t i = 0; i < k; i++)
        {
            retv.push_back(v[i].first);
        }

        return retv;
    }
};
