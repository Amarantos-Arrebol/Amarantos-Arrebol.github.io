迭代器

C++中面向对象有三大特性，封装、继承和多态，而迭代器就是封装思想的完美体现。它屏蔽了底层细节，让我们可以使用统一的方式来遍历容器，我们可以像用指针一样来遍历容器。迭代器之所以能实现像指针一样访问，主要是因为它重载了operator* 和 operator->。

void reserve(size_t n = 0)
{
    if (n > _capacity)
    {
        //开新空间
        char* tmp = new char[n + 1];//注意要多开一个

        //拷贝旧数据
        strcpy(tmp, _str);

        //释放旧空间,指向新空间
        delete[] _str;
        _str = tmp;

        //更新capacity
        _capacity = n;
    }
}

void reserve(size_t n)
{
	if (n > capacity())
	{
		size_t oldSize = size();
		T* tmp = new T[n];
		//第一次扩容可以不释放旧空间
		if (_start)
		{
			//memcpy(tmp, _start, sizeof(T) * oldSize);
			for (size_t i = 0; i < oldSize; i++)
			{
				tmp[i] = _start[i];
			}

			delete[] _start;//先调用析构函数清理空间，再调用operator delete 释放空间
		}

		_start = tmp;
		_finish = _start + oldSize;
		_endofstorage = _start + n;
	}
}

string& append(const char* s)
{
    //1.先判断是否需要开新空间
    size_t len = strlen(s);
    if (_capacity < _size + len)
    {
        //先考虑扩2倍，如果不够，则有多少扩多少
        size_t newCapacity = _capacity * 2;
        if (newCapacity < _size + len)
            newCapacity = _size + len;
        reserve(newCapacity);
    }

    //2.尾插字符串
    strcpy(_str + _size, s);
    _size += len;

    return *this;
}

string& insert(size_t pos, const char* str)
{
	assert(pos <= _size);

	//1.判断是否需要开空间
	size_t len = strlen(str);
	if (_size + len > _capacity)
	{
		//先考虑扩2倍，如果不够，则有多少扩多少
		size_t newCapacity = _capacity * 2;
		if (newCapacity < _size + len)
			newCapacity = _size + len;
		reserve(newCapacity);
	}
	
	//2.pos位置以及后面的字符往后挪
	//int end = _size;
	//while (end >= (int)pos)
	//{
	//	_str[end + len] = _str[end];
	//	--end;
	//}

	size_t end = _size + len;
	while (end >= pos + len)
	{
		_str[end] = _str[end - len];
		--end;
	}

	//3.插入字符串
	for (size_t i = 0; i < len; i++)
	{
		_str[pos + i] = str[i];
	}

	_size += len;

	return *this;
}

string& erase(size_t pos, size_t len = npos)
{
	assert(pos < _size);

	if (pos + len >= _size)
	{
		_str[pos] = '\0';
		_size = pos;
	}
	else
	{
		//hello hi

		size_t end = pos + len;

			_str[end - len] = _str[end];

		_size -= len;

	return *this;
}

size_t find(char c, size_t pos = 0)
{
	assert(pos < _size);

	for (size_t i = pos; i < _size; i++)
	{
		if (_str[i] == c)
			return i;
	}

	return npos;
}

size_t find(const char* s, size_t pos = 0) const
{
	assert(pos < _size);

	const char* ptr = strstr(_str + pos, s);
	if(ptr == nullptr)
		return npos;
	else
		return ptr - _str;
}

string substr(size_t pos, size_t len = npos) const
{
    //大于后面剩余串的长度，则直接取到结尾
    if (pos + len > _size)
    {
        len = _size - pos;
    }

    dyy::string sub;
    sub.reserve(len);
    for (size_t i = 0; i < len; i++)
    {
        sub += _str[pos + i];
    }

    return sub;
}

std::istream& operator>> (std::istream& is, string& str)
{
	str.clear();

	//先放到字符数组中，减少扩容的次数
	int i = 0;
	char buff[256];

	char ch;
	ch = is.get();
	while (ch != ' ' && ch != '\n')
	{
		buff[i++] = ch;
		if (i == 255)
		{
			buff[i] = '\0';
			str += buff;
			i = 0;
		}

		ch = is.get();
	}

	if (i > 0)
	{
		buff[i] = '\0';
		str += buff;
		i = 0;
	}

	return is;
}

#pragma once
#include <cassert>

namespace dyy
{
	template<class T>
	class vector
	{
	public:
		typedef T* iterator;

		vector()
		{}

		size_t size() const
		{
			return _finish - _start;
		}

		size_t capacity() const
		{
			return _endofstorage - _start;
		}
        
        bool empty() const
		{
			return _start == _finish;
		}
        
        ~vector()
        {
            if (_start)
            {
                delete[] _start;
                _start = _finish = _endofstorage = nullptr;
            }
        }

		T& operator[](size_t i)
		{
			assert(i < size());

			return _start[i];
		}
        
        const T& operator[](size_t i) const 
        {
            assert(i < size());

            return _start[i];
        }

	private:
		iterator _start = nullptr;
		iterator _finish = nullptr;
		iterator _endofstorage = nullptr;
	};
}

void reserve(size_t n)
{
	if (n > capacity())
	{
		size_t oldSize = size();
		T* tmp = new T[n];
		//第一次扩容可以不释放旧空间
		if (_start)
		{
			memcpy(tmp, _start, sizeof(T) * oldSize);
			delete[] _start;
		}

		_start = tmp;
		_finish = _start + oldSize;
		_endofstorage = _start + n;
	}
}

void push_back(const T& val)
{
	if (_finish == _endofstorage)
	{
		reserve(capacity() == 0 ? 4 : capacity() * 2);
	}

	*_finish = val;
	++_finish;
}

#pragma once
#include <cassert>

namespace dyy
{
	template<class T>
	class vector
	{
	public:
		typedef T* iterator;
		typedef const T* const_iterator;

		iterator begin()
		{
			return _start;
		}

		iterator end()
		{
			return _finish;
		}

		const_iterator begin() const
		{
			return _start;
		}

		const_iterator end() const
		{
			return _finish;
		}

		vector()
		{}

		vector(initializer_list<T> il)
			:_start(nullptr)
			, _finish(nullptr)
			, _endofstorage(nullptr)
		{
			reserve(il.size());

			for (auto& e : il)
			{
				push_back(e);
			}
		}

		vector(const vector<T>& v)
		{
			reserve(v.capacity());

			for (auto e : v)
			{
				push_back(e);
			}
		}

		//类模版的成员函数，也可以是一个函数模版
		template <class InputIterator>
		vector(InputIterator first, InputIterator last)
		{
			while (first != last)
			{
				push_back(*first);
				++first;
			}
		}

		vector(size_t n, const T& val = T())
		{
			reserve(n);
			for (size_t i = 0; i < n; i++)
			{
				push_back(val);
			}
		}

		vector(int n, const T& val = T())
		{
			reserve(n);
			for (size_t i = 0; i < n; i++)
			{
				push_back(val);
			}
		}

		//直接交换指针
		void swap(vector<T>& tmp)
		{
			std::swap(_start, tmp._start);
			std::swap(_finish, tmp._finish);
			std::swap(_endofstorage, tmp._endofstorage);
		}

		vector<T>& operator=(vector<T> v)
		{
			if (this != &v)
			{
				swap(v);
			}
			return *this;
		}

		size_t size() const
		{
			return _finish - _start;
		}

		size_t capacity() const
		{
			return _endofstorage - _start;
		}

		void clear()
		{

		}

		bool empty() const
		{
			return _start == _finish;
		}

		~vector()
		{
			if (_start)
			{
				delete[] _start;
				_start = _finish = _endofstorage = nullptr;
			}
		}

		T& operator[](size_t i)
		{
			assert(i < size());

			return _start[i];
		}

		const T& operator[](size_t i) const 
		{
			assert(i < size());

			return _start[i];
		}

		void reserve(size_t n)
		{
			if (n > capacity())
			{
				size_t oldSize = size();
				T* tmp = new T[n];
				//第一次扩容可以不释放旧空间
				if (_start)
				{
					//memcpy(tmp, _start, sizeof(T) * oldSize);
					for (size_t i = 0; i < oldSize; i++)
					{
						tmp[i] = _start[i];
					}

					delete[] _start;//先调用析构函数清理空间，再调用operator delete 释放空间
				}

				_start = tmp;
				_finish = _start + oldSize;
				_endofstorage = _start + n;
			}
		}

		void push_back(const T& val)
		{
			/*if (_finish == _endofstorage)
			{
				reserve(capacity() == 0 ? 4 : capacity() * 2);
			}

			*_finish = val;
			++_finish;*/

			insert(_finish, val);
		}

		void pop_back()
		{
			assert(!empty());

			--_finish;
		}

		//在pos位置插入一个值
		//在pos位置插入一段迭代器区间
		//在pos位置插入n个值
		iterator insert(iterator pos, const T& val)
		{
			assert(pos >= _start && pos <= _finish);
			
			//1.看需不需要扩容
			if (_finish == _endofstorage)
			{
				size_t len = pos - _start; // 记录pos到起始位置的距离
				reserve(capacity() == 0 ? 4 : capacity() * 2);
				pos = _start + len; // 更新pos的位置
			}
			//2.挪数据
			iterator i = _finish;
			while (i > pos)
			{
				*i = *(i - 1);
				--i;
			}
			//3.放要插入的值
			*pos = val;
			++_finish;

			return pos;
		}

		iterator erase(iterator pos)
		{
			assert(pos >= _start && pos < _finish);

			//pos后面的数据往前挪
			iterator i = pos + 1;
			while (i < _finish)
			{
				*(i - 1) = *i;
				++i;
			}
			--_finish;

			return pos;
		}

		void resize(size_t n, T val = T())
		{
			if (n <= size())
			{
				_finish = _start + n;
			}
			else //n大于当前size，扩容并插入值
			{
				reserve(n);
				
				while (_finish < (_start + n))
				{
					*_finish = val;
					++_finish;
				}
			}
		}

	private:
		iterator _start = nullptr;
		iterator _finish = nullptr;
		iterator _endofstorage = nullptr;
	};
}
